import 'package:flutter/material.dart';

void main() {
  runApp(FitnessApp());
}

class FitnessApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: HomeScreen(),
    );
  }
}

class HomeScreen extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Fitness App'),
      ),
      drawer: AppDrawer(),
      body: Center(
        child: Text('Welcome to the Fitness App!'),
      ),
    );
  }
}

class AppDrawer extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Drawer(
      child: ListView(
        padding: EdgeInsets.zero,
        children: [
          DrawerHeader(
            decoration: BoxDecoration(
              color: Colors.blue,
            ),
            child: Text(
              'Fitness App',
              style: TextStyle(
                color: Colors.white,
                fontSize: 24,
              ),
            ),
          ),
          ListTile(
            leading: Icon(Icons.fitness_center),
            title: Text('Workouts'),
            onTap: () {
              // Navigate to the Workouts screen
              Navigator.pop(context); // Close the drawer
              Navigator.push(
                context,
                MaterialPageRoute(
                  builder: (context) => PageScreen('Workouts'),
                ),
              );
            },
          ),
          ListTile(
            leading: Icon(Icons.show_chart),
            title: Text('Progress'),
            onTap: () {
              // Navigate to the Progress screen
              Navigator.pop(context); // Close the drawer
              Navigator.push(
                context,
                MaterialPageRoute(
                  builder: (context) => PageScreen('Progress'),
                ),
              );
            },
          ),
          ListTile(
            leading: Icon(Icons.settings),
            title: Text('Settings'),
            onTap: () {
              // Navigate to the Settings screen
              Navigator.pop(context); // Close the drawer
              Navigator.push(
                context,
                MaterialPageRoute(
                  builder: (context) => PageScreen('Settings'),
                ),
              );
            },
          ),
        ],
      ),
    );
  }
}

class PageScreen extends StatelessWidget {
  final String pageTitle;

  PageScreen(this.pageTitle);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(pageTitle),
      ),
      body: Center(
        child: Text(
          'Welcome to the $pageTitle page!',
          style: TextStyle(fontSize: 20.0),
        ),
      ),
    );
  }
}
