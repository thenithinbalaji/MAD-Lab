import 'package:flutter/material.dart';

void main() {
  runApp(RestaurantApp());
}

class RestaurantApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: MenuScreen(),
    );
  }
}

class MenuScreen extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Restaurant Menu'),
      ),
      body: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          Padding(
            padding: const EdgeInsets.all(16.0),
            child: Text(
              'Categories',
              style: TextStyle(
                fontSize: 20.0,
                fontWeight: FontWeight.bold,
              ),
            ),
          ),
          CategoryList(), // Display categories vertically
          Padding(
            padding: const EdgeInsets.all(16.0),
            child: Text(
              'Menu Items',
              style: TextStyle(
                fontSize: 20.0,
                fontWeight: FontWeight.bold,
              ),
            ),
          ),
          Expanded(
            child: MenuItemList(), // Display menu items horizontally
          ),
        ],
      ),
    );
  }
}

class CategoryList extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        CategoryItem('Appetizers'),
        CategoryItem('Main Course'),
        CategoryItem('Desserts'),
        // Add more categories as needed
      ],
    );
  }
}

class CategoryItem extends StatelessWidget {
  final String categoryName;

  CategoryItem(this.categoryName);

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.symmetric(vertical: 8.0),
      child: Text(
        categoryName,
        style: TextStyle(fontSize: 16.0),
      ),
    );
  }
}

class MenuItemList extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Row(
      children: [
        MenuItem('Pizza'),
        MenuItem('Burger'),
        MenuItem('Salad'),
        // Add more menu items as needed
      ],
    );
  }
}

class MenuItem extends StatelessWidget {
  final String itemName;

  MenuItem(this.itemName);

  @override
  Widget build(BuildContext context) {
    return Container(
      margin: EdgeInsets.all(8.0),
      padding: EdgeInsets.all(16.0),
      decoration: BoxDecoration(
        border: Border.all(color: Colors.grey),
        borderRadius: BorderRadius.circular(8.0),
      ),
      child: Text(
        itemName,
        style: TextStyle(fontSize: 16.0),
      ),
    );
  }
}
